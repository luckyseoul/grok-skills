# /Applications/Safari.app/Contents/MacOS/Safari --args file:///Users/user/hitting_candidates/drive_hit_....png

# lldb -p $(pgrep Safari) -n Safari

# 3. Break on the exact vulnerable write path (from source analysis)
breakpoint set --name PNGImageDecoder::rowAvailable
# or more precise:
# breakpoint set --file PNGImageDecoder.cpp --line 507   # the interlaceBuffer indexing line
# breakpoint set --file PNGImageDecoder.cpp --line 509   # row = interlaceBuffer + (rowIndex * ...

# 4. When hit, inspect for OOB
# (lldb) p rowIndex
# (lldb) p size().height()
# (lldb) p colorChannels
# (lldb) p size().width()
# (lldb) p (char*)interlaceBuffer
# (lldb) p/x (uint64_t)(row - interlaceBuffer)   # offset into buffer; if >> allocated size -> OOB
# (lldb) p buffer.backingStore()   # the destination write target

# 5. Watch for the write
# (lldb) watchpoint set expression -- (char*)destinationRow.data()   # or the address var
